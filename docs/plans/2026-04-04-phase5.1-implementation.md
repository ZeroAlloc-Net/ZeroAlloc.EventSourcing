# Phase 5.1: Snapshot-Optimized Repository Loading — Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development to implement this plan task-by-task.

**Goal:** Enable aggregate loading to skip old events by using snapshots, reducing replay overhead for long-lived streams.

**Architecture:** Decorator pattern wrapping `AggregateRepository<TAggregate, TId>`. The decorator handles snapshot lookup and position-aware event replay without modifying the core repository. `ISnapshotLoadingStrategy` enum allows configurable handling of edge cases (trust snapshot, validate, or ignore).

**Tech Stack:** Existing `ISnapshotStore<TState>` from Phase 5, `IEventStore`, C# record structs, ValueTask, Roslyn (for generated helpers if needed).

```csharp
namespace ZeroAlloc.EventSourcing;

/// <summary>
/// Configurable strategy for handling snapshots during aggregate load.
/// </summary>
public enum SnapshotLoadingStrategy
{
    /// <summary>
    /// Trust the snapshot completely. Replay events from snapshot position onward.
    /// Fastest, but assumes snapshot is always valid. Use when snapshots are immutable.
    /// </summary>
    TrustSnapshot = 0,

    /// <summary>
    /// Validate snapshot position exists in event store before using.
    /// If position not found, fall back to replaying from start.
    /// Safer, but adds validation overhead.
    /// </summary>
    ValidateAndReplay = 1,

    /// <summary>
    /// Ignore snapshots entirely. Always replay from start.
    /// Useful for disabling snapshot optimization without code changes.
    /// </summary>
    IgnoreSnapshot = 2
}
```

**Step 2: Implement SnapshotCachingRepositoryDecorator**

```csharp
using ZeroAlloc.EventSourcing;
using ZeroAlloc.Results;

namespace ZeroAlloc.EventSourcing.Aggregates;

/// <summary>
/// Decorator that wraps <see cref="IAggregateRepository{TAggregate, TId}"/> to optimize loading
/// using snapshots. Checks for snapshots before replaying events, reducing read amplification.
/// </summary>
public sealed class SnapshotCachingRepositoryDecorator<TAggregate, TId, TState>
    : IAggregateRepository<TAggregate, TId>
    where TAggregate : IAggregate
    where TId : struct
    where TState : struct, IAggregateState<TState>
{
    private readonly IAggregateRepository<TAggregate, TId> _innerRepository;
    private readonly ISnapshotStore<TState> _snapshotStore;
    private readonly SnapshotLoadingStrategy _strategy;
    private readonly Func<TAggregate, TState> _getState;
    private readonly Action<TAggregate, TState, StreamPosition> _setState;
    private readonly Func<TId, StreamId> _streamIdFactory;
    private readonly IEventStore _eventStore;

    public SnapshotCachingRepositoryDecorator(
        IAggregateRepository<TAggregate, TId> innerRepository,
        ISnapshotStore<TState> snapshotStore,
        SnapshotLoadingStrategy strategy,
        Func<TAggregate, TState> getState,
        Action<TAggregate, TState, StreamPosition> setState,
        Func<TId, StreamId> streamIdFactory,
        IEventStore eventStore)
    {
        ArgumentNullException.ThrowIfNull(innerRepository);
        ArgumentNullException.ThrowIfNull(snapshotStore);
        ArgumentNullException.ThrowIfNull(getState);
        ArgumentNullException.ThrowIfNull(setState);
        ArgumentNullException.ThrowIfNull(streamIdFactory);
        ArgumentNullException.ThrowIfNull(eventStore);

        _innerRepository = innerRepository;
        _snapshotStore = snapshotStore;
        _strategy = strategy;
        _getState = getState;
        _setState = setState;
        _streamIdFactory = streamIdFactory;
        _eventStore = eventStore;
    }

    public async ValueTask<Result<TAggregate, StoreError>> LoadAsync(
        TId id,
        CancellationToken ct = default)
    {
        var streamId = _streamIdFactory(id);

        // Check for snapshot based on strategy
        if (_strategy != SnapshotLoadingStrategy.IgnoreSnapshot)
        {
            var snapshot = await _snapshotStore.ReadAsync(streamId, ct).ConfigureAwait(false);
            if (snapshot.HasValue)
            {
                // Load aggregate from inner repository
                var result = await _innerRepository.LoadAsync(id, ct).ConfigureAwait(false);
                if (!result.IsSuccess)
                    return result;

                // Restore snapshot state
                var aggregate = result.Value;
                _setState(aggregate, snapshot.Value.State, snapshot.Value.Position);

                // Replay events from snapshot position onward
                await ReplayFromPositionAsync(aggregate, streamId, snapshot.Value.Position.Next(), ct).ConfigureAwait(false);

                return Result<TAggregate, StoreError>.Success(aggregate);
            }
        }

        // No snapshot or strategy ignores it — delegate to inner repository
        return await _innerRepository.LoadAsync(id, ct).ConfigureAwait(false);
    }

    public ValueTask<Result<AppendResult, StoreError>> SaveAsync(
        TAggregate aggregate,
        TId id,
        CancellationToken ct = default)
    {
        // Delegate to inner repository (decorator doesn't handle saving)
        return _innerRepository.SaveAsync(aggregate, id, ct);
    }

    private async Task ReplayFromPositionAsync(
        TAggregate aggregate,
        StreamId streamId,
        StreamPosition fromPosition,
        CancellationToken ct)
    {
        await foreach (var envelope in _eventStore.ReadAsync(streamId, fromPosition, ct).ConfigureAwait(false))
        {
            aggregate.ApplyHistoric(envelope.Event, envelope.Position);
        }
    }
}
```
